Promisifying

          
					
// JavaScript snippet that converts a
// callback-based function to a 
// promise-based function

const promisifyTimout = (seconds) => {
  return new Promise(function (resolve) {
    setTimeout(resolve, seconds * 1000);
  });
};

function btnClickHandler1() {
  promisifyTimout(3).then(() => {
    console.log('Promisify setTimeout 
      - Wait 3 seconds');

  });
}
